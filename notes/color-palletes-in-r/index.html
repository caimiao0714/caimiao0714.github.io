# devtools::install_github("disarm-platform/MapPalettes")
pacman::p_load(dplyr, ggplot2, MapPalettes)
mtcars %>% 
  filter(carb <= 4) %>% 
  ggplot(aes(mpg, gear, color = factor(carb))) + 
  geom_point(size = 4) + theme_classic() + 
  scale_color_manual(values = 
      MapPalettes::map_palette("tealberry_pie", 4))
